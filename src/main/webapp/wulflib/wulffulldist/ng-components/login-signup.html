require(['angular', 'wulf-ng/login/login'], function (angular) {
angular.module('wf.ng.demo', ['wf.angular.login']).controller('loginDemoController', loginDemoController);
function loginDemoController() {
    //Login account sign-up:
    this.signuplogin = {};
    this.signuplogin.loginType = 'signup';
    this.signuplogin.titleText = 'BTS Site Manager';
    this.signuplogin.version = 'Version 0.0.1 beta';
    this.signuplogin.copyright = '2016 Nokia. All rights reserved.';
    this.signuplogin.onLogin = function (username, password) {
        alert('This is an on-login callback.\n username : '+ username +'\n password : ' + password);
    };
    this.signuplogin.forgotPsw = function () {
        alert('This is a forget password callback.');
    };
    this.signuplogin.onSignUp = function () {
        alert('This is a sign up callback.');
    };
}
