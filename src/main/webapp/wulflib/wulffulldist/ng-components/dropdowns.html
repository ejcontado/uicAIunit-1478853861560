Pull down(Dropdown)




Overview

Angular version of dropdown component. It contains two components wfDropdown and wfDropdownItem which are implemented by using .component() method introduced since Angular1.5.

wf-dropdown settings

Attributes:Bindings Accept value Notes
pattern: <
  • standard (Default)
  • small
  • dark
 The pattern of dropdown list.
disable: <?
  • false (Default)
  • true
 Dropdown list status

wf-dropdown-item settings

Attributes:Bindings Accept value Notes
value: < String Dropdown list item value
label: < String(Optional) Dropdown list item display text. If not present, display text is binded by value
disable: <?
  • false (Default)
  • true
 Dropdown list item status
select: <?
  • false (Default)
  • true
 Select dropdown list item as default
on-item-selected: <? function(value, label) Callback function when pulldown item was selected.
  • value - current dropdown list item value
  • label - current dropdown list item display text
  • event - event when item is selected

Keyboard accessibility

Besides the basic keyboard support, e.g. Tab, Shift + Tab and Enter. This component supports the following keystrokes for the interaction.
Interaction Keystrokes Notes
Move focus to next/previous dropdown item ↓ or ↑ - next/previous dropdown item
  • The disabled dropdown item would be skipped
Select dropdown item Spacebar/Enter - Select current focused item
  • The disabled item cannot be selected

Missing functionality

  • Scrollbar not support yet.
  • Tooltip not support yet.
                    



    
        
        
        
        
    






    
        
        
        
        
    






    
        
        
        
        
    


                    
                
                    
angular.module('wf.ng.demo', ['wf.angular.dropdowns','wf.angular.tabs']).controller('DropdownsDemoController', DropdownsDemoController);
function DropdownsDemoController() {
    var newPullDownInstance = function (type, disable) {
        return {
            type: type,
            disable: disable
        }
    };

    var newPullDownItemInstance = function (value, label, disable, select, onItemSelected) {
        return {
            value: value,
            label: label,
            disable: disable,
            select: select,
            onItemSelected: onItemSelected
        }
    };

    this.standardPullDownInstance = newPullDownInstance("standard", false);
    this.standardDisableItemsPullDownInstance = newPullDownInstance("standard", false);
    this.smallPullDownInstance = newPullDownInstance("small", false);
    this.darkPullDownInstance = newPullDownInstance("dark", false);
    this.darkDisableItemsPullDownInstance = newPullDownInstance("dark", false);

    var onItemSelected = function(value, label) {
        alert('value : '+ value + '\n' +'label: ' + label );
    };
    this.items = [
        newPullDownItemInstance('One', 'One option', false, false, onItemSelected),
        newPullDownItemInstance('Two', 'Two option', false, true, onItemSelected),
        newPullDownItemInstance('Three', 'Three option', false, false, onItemSelected),
        newPullDownItemInstance('Four', 'Very long long long long long item', false, false, onItemSelected)
    ];
    this.disableItems = [
        newPullDownItemInstance('One', 'First', false, true, onItemSelected),
        newPullDownItemInstance('Two', 'Second', true, false, onItemSelected),
        newPullDownItemInstance('Three', 'Third', true, false, onItemSelected),
        newPullDownItemInstance('Four', 'Fourth', false, false, onItemSelected)
    ];
}
angular.element(document).ready(function () {
    angular.bootstrap(document, ['wf.ng.demo'], {
        strictDi: true
    });
