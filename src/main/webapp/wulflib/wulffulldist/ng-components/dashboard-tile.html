Dashboard tile

Overview

Dashboard tile component is implemented by using .component() method introduced since Angular1.5.

wf-dashboard-tile settings

Attributes:Bindings Accept value Note
tileTitle: < String (Default: null) The title of the tile.
tileStatus: <
  • true (Default)
  • false
 The block status of the tile. True is unblock and false is blocked.
tileFrq: < String (Default: null) The frequency of the tile.
tileId: < String (Default: null) The Id of the tile.
pieData: < Integer: 0-100 (Default: 0) The data of the tile's pie. 
                    
    

            
            
            
            
    

                    
                
    
    angular.module('wf.ng.demo', ['wf.angular.dashboardTile']).controller('tileDemoController', tileDemoController);
    function tileDemoController(){
        var newInstance = function (tileTitle, tileStatus, tileFrq, tileId, pieData) {
          return {
              tileTitle: tileTitle,
              tileStatus: tileStatus,
              tileFrq: tileFrq,
              tileId: tileId,
              pieData: pieData
          }
        };
        var self = this;
        self.tileData1 = newInstance('WCDMA', true, 10, 'LCELL-3102',33);
        self.tileData2 = newInstance('LTE', false, 10, 'LCELL-3102',90);
        self.tileData3 = newInstance('LTE', true, 10, 'LCELL-3102',100);
        self.tileData4 = newInstance('WCDMA', true, 687, 'LCELL-3102',0);

    }
