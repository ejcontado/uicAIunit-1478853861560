Overview

Angular version of progress component which is implemented by using .component() method introduced since Angular1.5.

wf-progress settings

Attributes:Bindings Accept value Note
pattern: <
  • bar - standard progress bar (Default)
  • sm-bar - small progress bar
  • indeterminate-bar - indeterminate progress bar
  • loading - standard progress loading icon
  • sm-loading - small progress loading icon
 The pattern of progress bar to be created.
percentage: < Integer: 0-100 (Default: 0) The current value of progress completed.
show-value: <
  • end - Put the value indicator to end of the bar
  • inline - Put the value indicator on the bar
  • null - Default
 The position where to put the value indicator.
active: < Boolean: true or false (Default) The active status of indeterminate progress bar.

If you are using this component in CSP mode, please make sure the 'unsafe-inline' is enabled.

    angular.module('wf.ng.demo', ['wf.angular.progress', 'wf.angular.tabs']).controller('ProgressDemoController', ProgressDemoController);

    function ProgressDemoController(){
        this.standardbar = {};
        this.smallbar = {};
        this.indeterminatebar = {};
        this.loading = {};
        this.smallLoading = {};

        this.standardbar.pattern = 'bar';
        this.standardbar.emptyValue = 0;
        this.standardbar.inProgressValue = 60;
        this.standardbar.completedValue = 100;
        this.standardbar.showValueInline = 'inline';
        this.standardbar.showValueInEnd = 'end';

        this.smallbar.pattern = 'sm-bar';
        this.smallbar.inProgressValue = 60;

        this.indeterminatebar.pattern = 'indeterminate-bar';
        this.indeterminatebar.isActive = true;

        this.loading.pattern = 'loading';
        this.smallLoading.pattern = 'sm-loading';

        this.random = function(){
            var value = Math.floor(Math.random() * 100 + 1);
            this.standardbar.inProgressValue = value;
            this.smallbar.inProgressValue = value;
            value > 50 ? this.indeterminatebar.isActive = true : this.indeterminatebar.isActive = false;
        }
    }
