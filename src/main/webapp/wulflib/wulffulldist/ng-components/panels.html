Overview

Angular version of panel component. It contains three components wfPanels, wfPanelHeader and wfPanelSection which are implemented by using .component() method introduced since Angular1.5. This component can be used to generate section in panel body from both static and dynamic model data (using ng-repeat directive for wf-panel-section). wfPanelHeader is optional. If you want to customize the styles for panel header, you can use wfPanelHeader component and that will let header attribute invalid.

wf-panels settings

Attributes:Bindings Arguments Note
slide-direction: < Default:vertical Arguments:vertical/horizontal Whether panel collapse/expand in horizontal.
slide-animation: < Default:false Arguments:true/false Whether panel support collapse/expand animation
header: @ string title of panel
on-toggle: & function (isShow ,header, event) Callback function when toggle panel.
  • isShow - whether current panel is show or hide
  • header - title of current panel
  • event - event when toggle panel
scrollbar: < object Config for scrollbar. Refers to scroll component scrollbar-config
  • e.g. { height: '300px' }

Keyboard accessibility

Besides the basic keyboard support, e.g. Tab,Enter. This component also supports the following keystrokes for the interaction.
Interaction Keystrokes Notes
collapse/expand panel body Space
  • Before to collapse/expand panel body, the icon placed section must be focus firstly

angular.module('wf.ng.demo', ['wf.angular.constants', 'wf.angular.panels','wf.angular.tabs']).controller('SectionDemoController', SectionDemoController);
function SectionDemoController() {
    var makeSection = function (title, divClass, content) {
        return {
            title: title,
            divClass: divClass,
            content: content
        }
    };
    var self = this;
    self.sections = [makeSection('Title of section', 'dummy-picture', 'Aenean eu leo quam. Pellentesque ornare sem lacinia quam venenatis vestibulum. Lorem ipsum dolor.'), makeSection('Title of section', 'dummy-picture', 'Aenean eu leo quam. Pellentesque ornare sem lacinia quam venenatis vestibulum. Lorem ipsum dolor.')];
    self.onToggle = function (isShow ,header) {
        var state = isShow?'hidden':'shown';
        alert(header + ' will be ' + state);
    };
    self.config = {
        height: '300px',
        onScroll: function() {
                    var scrollObj = angular.element(this);
                        alert('This is an onScroll callback. \n scroll direction: ' + scrollObj[0].mcs.direction +'\n' +
                                'offset left: ' + scrollObj[0].mcs.left +'\n' +
                                'offset top: ' + scrollObj[0].mcs.top );
                  }
    };
}
angular.element(document).ready(function () {
    angular.bootstrap(document, ['wf.ng.demo'], {
        strictDi: true
    });
});
