angular.module('wf.ng.demo', ['wf.angular.tabs']).controller('TextareaDemoController', TextareaDemoController).controller('DemoCtrl', function(){});
function TextareaDemoController() {
    var newInstance = function (disable, readonly, content) {
        return {
            disable: disable,
            readonly: readonly,
            content: content
        }
    };

    this.textareaInstance = newInstance(true, true, "TEXT TEXT TEXT TEXT TEXT TEXT TEXT TEXT TEXT TEXT TEXT TEXT TEXT TEXT TEXT");

    this.enableDisable = "Enable";
    this.updateEnableDisable = function () {
        this.textareaInstance.disable = !this.textareaInstance.disable;
        if(this.textareaInstance.disable) {
            this.enableDisable = "Enable";
        } else {
            this.enableDisable = "Disable";
        }
    };

    this.readonlyOperation = "Normal";
    this.updateReadonly = function () {
        this.textareaInstance.readonly = !this.textareaInstance.readonly;
        if(this.textareaInstance.readonly) {
            this.readonlyOperation = "Normal";
        } else {
            this.readonlyOperation = "Readonly";
        }
    };
}
