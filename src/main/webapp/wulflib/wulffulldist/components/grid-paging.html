The next step is to create a DIV element within the body of the html document.


    <div id="table-paging"></div>

The last step is to initialize the widget. 


    <script>
        $(document).ready( function() {
        /*---------------- test data ----------------*/
        var data = new Array();
        data[0] = {
        name: 'Avatar',
        alarmnumber: '113211',
        alarmtext: 'System module error',
        alarmtime: "1985/03/26"
        };
        ...

        var source =
        {
        localdata: data,
        datafields:
        [
        {name: 'name', type: 'string'},
        {name: 'alarmnumber', type: 'number'},
        {name: 'alarmtext', type: 'string'},
        {name: 'alarmtime', type: 'string'}
        ],
        datatype: "array"
        };

        var columns = [
        {text: 'Distinguished Name',datafield: 'name'},
        {text: 'Alarm Number', datafield: 'alarmnumber',cellsalign: 'right', align:"right"},
        {
        text: 'Alarm Text', datafield: 'alarmtext', width: 200},
        {
        text: 'Alarm Time', datafield: 'alarmtime'}
        ]

        var dataAdapter = new $.jqx.dataAdapter(source);

        $("#table-paging").jqxGrid(
        {
            width: 700,
            height:330,
            source: dataAdapter,
            showfilterrow: true,
            selectionmode: 'multiplecellsextended',
            columns: columns,
            scrollBarSize: 8,
            rowsheight: 26,
            columnsheight: 26,
            pageable: true,
            pagerheight: 40,
            pagerrenderer: $.grid.pagerrenderer.bind(null, "#table-paging"),
            
            handlekeyboardnavigation: function (e) {
                return $.grid.handlekeyboardnavigation(e);
            }
            
        });
    </script>
