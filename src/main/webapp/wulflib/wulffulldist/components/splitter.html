For splitter components, import following css files:
<link href="../wulfdist/css/jqx.base.css" rel="stylesheet"/>

To use the splitter, see more details from the jqxSplitter documentation. 

<!-- Standard splitter (from jqxWidgets) -->
<div id="mySplitterVert" class="n-splitter">
    <div>Area 1</div>
    <div>Area 52</div>
</div>

<script src="../wulfdist/js/jquery.js"></script>
<script src="../wulfdist/js/jqxcore.js"></script>
<script src="../wulfdist/js/jqxsplitter.js"></script>

<script>
var panels = [{size: 100, collapsible: false}, {collapsible: false}];

<!-- Vertical splitter -->
$('#mySplitterVert').jqxSplitter({height: 200, width: 300, panels: panels});

<!-- Horizontal splitter -->
$('#mySplitterHoriz').jqxSplitter({height: 200, width: 300, orientation: 'horizontal', panels: panels});
</script>
