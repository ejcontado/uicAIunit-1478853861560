    <script src="../wulfdist/js/dependencies/jquery.min.js"></script>
    <script>
    $(document).ready(function () {

        var $lifecycle = $("#lifecycle1");
        $(".next").on("click", $lifecycle, function () {
            $lifecycle.find(".n-lifecycle").each(function () {
                if (!$(this).hasClass("selected")) {
                    $(this).addClass("selected");
                    return false;
                }
            });
        });

        $(".previous").on("click", $lifecycle, function () {
            $($lifecycle.find(".n-lifecycle").get().reverse()).each(function () {
                if ($(this).hasClass("selected")) {
                    $(this).removeClass("selected");
                    return false;
                }
            });
        });
    });
    </script>
