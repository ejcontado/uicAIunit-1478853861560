	<script>
	$(document).ready( function ()
	{
		$('#myTreeVerticalNav').tree({
			dataSource: dataSource,
			folderSelect :false
		});
		<-- Insert data for tree component -->
		dataSource = function(parentData, callback){//...}

		<-- Optionally define static data for tree component -->
		var childNodesArray = [
			{ "name": "Add WULF", "type": "item", "attr":{"parentNode" : "rootNode", "src" :"#", "id":"start_add"}}
		];
		function staticDataSource(openedParentData, callback) {
			if(typeof openedParentData.attr != 'undefined'){
				callback({
					data: get(childNodesArray,openedParentData.name)
				});
			} else {
				callback({
					data: get(childNodesArray ,"rootNode")
				});
			}
		}
	});
	</script>
    To disable the tree, you can simply call disable method for the tree component:

    Option 2 - Simple with pure navbar component (when there is no tree-like menu structure needed): 
    
	<h2 class="secondary-vertical-heading">Navigation title text</h2>
	<div class="panel panel-shadow nav-secondary-vertical">
		<nav class="navbar" role="navigation">

				</li>

					<span class="icon icon-bullet"><a><span>System Req</span</a></span>
				</li>

	</div>
    Javascript for the Option 2 to work: 
    
	<script>
	$('.nav li').click(function(e) {
		$('.nav li.active').removeClass('active');
		var $this = $(this);
		if (!$this.hasClass('active')) {
			$this.addClass('active');
		}
		e.preventDefault();
	});
	</script>
