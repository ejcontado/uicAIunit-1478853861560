JS & CSS dependency:

For pull down with scrollbar components,you can use following css import code:
<link href="../wulfdist/css/wulf.basic.min.css" rel="stylesheet"/>
<link href="../wulfdist/css/wulf.buttons.min.css" rel="stylesheet">
<link href="../wulfdist/css/wulf.dropdowns.min.css" rel="stylesheet">
<link href="../wulfdist/css/wulf.tooltip.min.css" rel="stylesheet">

    <!-- Basic Pull down -->
    <div class="btn-group selectlist selectlist-resize" data-resize="auto" data-initialize="selectlist" id="mySelectlist">
    <button class="btn btn-default dropdown-toggle" data-toggle="dropdown" type="button">
    <span class="selected-label"></span>
    <span class="selected-caret" ><span class="caret"></span></span>
    <span class="sr-only">Toggle Dropdown</span>
    </button>
    <ul class="dropdown-menu n-dropdown-menu-scroll" role="menu">
    <li data-value="1"><a href="#"><span>One</span></a></li>
    <li data-value="2"><a href="#"><span>Two</span></a></li>
    </ul>
    <input class="hidden hidden-field" name="mySelectlist" readonly="readonly" aria-hidden="true" type="text"/>
    </div>

    <!-- JS-script needed -->
    <script src="../wulfdist/js/dependencies/fuelux.js" type="text/javascript"></script>
    <script>
        $('#mySelectlist').adaptiveSelectlist();
        $(document).ready(function(){
            $(".n-dropdown-menu-scroll").nScrollbar();
        })
    </script>

    <!-- The default value of pull down menu height is 150px. You can change the height by definition of follow style -->
    .n-dropdown-menu-scroll{
        height: 200px !important;
    }
