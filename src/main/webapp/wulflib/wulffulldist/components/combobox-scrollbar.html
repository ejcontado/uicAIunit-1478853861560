<link href="../wulfdist/css/wulf.buttons.min.css" rel="stylesheet"/<

<link href="../wulfdist/css/wulf.inputfield.min.css" rel="stylesheet"/<

    <!-- Combobox scroll (textfield) -->
    <label for="combo1" class="control-label">Standard:</label>
    <div class="input-group input-append dropdown combobox" data-initialize="combobox" id="myCombobox">
        <input id="combo1" type="text" class="form-control n-inputfield">
        <div class="input-group-btn">
            <button type="button" class="btn btn-default dropdown-toggle" data-toggle="dropdown"><span class="caret"></span></button>
            <ul class="dropdown-menu dropdown-menu-right n-dropdown-menu-scroll>
                <li data-value="1"><a href="#"><span>One</span></a></li>
                <li data-value="2"><a href="#"><span>Two</span></a></li>
            </ul>
        </div>
    </div>

    <!-- Combobox scroll filter(textfield) -->
    <label for="comboFilter" class="control-label">Standard:</label>
    <div class="input-group input-append dropdown combobox combobox-filter" data-initialize="combobox" id="myComboboxFilter">
        <input id="comboFilter" type="text" class="form-control n-inputfield">
        <div class="input-group-btn">
            <button type="button" class="btn btn-default dropdown-toggle" data-toggle="dropdown"><span class="caret"></span></button>
            <ul class="dropdown-menu dropdown-menu-right n-dropdown-menu-scroll>
                <li data-value="1"><a href="#"><span>One</span></a></li>
                <li data-value="2"><a href="#"><span>Two</span></a></li>
            </ul>
        </div>
    </div>

    <!-- JS-script needed -->
    <script src="../wulfdist/js/dependencies/fuelux.js" type="text/javascript"></script>
    <script>
        $('#myCombobox').combobox();
        $(document).ready(function () {
            $('ul.dropdown-menu.dropdown-menu-right li').click(function () {
                $('ul.dropdown-menu.dropdown-menu-right li').removeClass('selected');
                $(this).addClass('selected');
            });
            $(".n-dropdown-menu-scroll").nScrollbar();
        });
    </script>

    <!-- The default value of combobox menu height is 150px. You can change the height by definition of follow style -->
    .n-dropdown-menu-scroll{
        height: 200px !important;
    }
