Place the chart into a <div>

<div id="chartContainer" style="width:600px; height:600px">
</div>

Data and Settings ( For more settings, see the jqxChart API Reference )


// the data to show on the chart
var sampleData = [
    {'Phone':'Model J','Share':15},
    {'Phone':'Model I','Share':4},
    {'Phone':'Model H','Share':4},
    {'Phone':'Model G','Share':15},
    {'Phone':'Model F','Share':16},
    {'Phone':'Model E','Share':7},
    {'Phone':'Model D','Share':28},
    {'Phone':'Model C','Share':3},
    {'Phone':'Model B','Share':4},
    {'Phone':'Model A','Share':5}
];
// the settings of the chart
var settings = {
    title: '',
    description: '',
    enableAnimations: true,
    showLegend: false,
    showBorderLine: false,
    backgroundColor:'transparent',
    source: sampleData,
    seriesGroups:
        [
            {
                type: 'donut',
                showLabels: true,
                toolTipFormatFunction: function (value, itemIndex) {
                    var browser =  sampleData[itemIndex]['Phone'];
                    return browser + '  <b>' + value + '</b>';
                },
                series:
                    [
                        {
                            dataField: 'Share',
                            displayText: 'Phone',
                            labelRadius: 175,
                            initialAngle: 90,
                            radius: 165,
                            innerRadius: 100,
                            centerOffset: 0,
                            formatFunction : function (value, itemIndex) {
                                var browser =  sampleData[itemIndex]['Phone'];
                                return browser + ' ' + value + '%';
                            }
                        }
                    ]
            }
        ]
};
// setup the chart
$('#chartContainer').jqxChart(settings);
$('#chartContainer').jqxChart('addColorScheme', 'nokiaScheme', ['#00F04D','#DB6D00','#D8D9DA','#68717A','#FFCC00','#474747','#00C9FF','#124191','#B3B8BC','#0B6DAF']);
$('#chartContainer').jqxChart('colorScheme', 'nokiaScheme');
