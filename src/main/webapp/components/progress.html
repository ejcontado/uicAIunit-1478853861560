The sr-only class setting for the span hides the percentage showing. The big ok icon is made with the help of bigger heading setting.


    <!-- Progress without value -->
    <div class="progress">
        <div class="progress-bar" id="myid1" role="progressbar" aria-valuenow="40" aria-valuemin="0" aria-valuemax="100"
        style="width: 40%;">
            <span class="sr-only">40% Complete</span>
        </div>
    </div>

    <!-- Progress with value (OT3 style) -->
    <div class="progress">
        <div class="progress-bar" id="myid2" role="progressbar" aria-valuenow="40" aria-valuemin="0" aria-valuemax="100"
        style="width: 40%;">
            <span class="progress-value-inverse">40%</span>
        </div>
    </div>

    <!-- Small -->
