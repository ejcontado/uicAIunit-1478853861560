Bar Chart (jqWidget)


JS & CSS dependency:

For bar chart(jqWidget) components, it will be ok to import following css file:
<link  href="../wulfdist/css/jqx.base.css" rel="stylesheet"/>
<link href="../wulfdist/css/wulf.basic.min.css" rel="stylesheet"/>
<link href="../wulfdist/css/wulf.chart.min.css" rel="stylesheet"/>
<link href="../wulfdist/css/wulf.panel.min.css" rel="stylesheet"/>

Place the chart into a <div> 


<!-- Main chart div -->
<div id="chartContainer" style="width:600px; height:400px">
</div>

Data and Settings ( For more settings, see the jqxChart API Reference )


// the data to show on the chart
var sampleData = [
    {Voice: '0m', Data: "14"},
    {Voice: '25m', Data: "15"},
    {Voice: '50m', Data: "18"},
    {Voice: '100m', Data: "5.4"},
    {Voice: '125m', Data: "2.7"},
    {Voice: '150m', Data: "4"},
    {Voice: '200m', Data: "2"},
    {Voice: '250m', Data: "6"},
    {Voice: '275m', Data: "9"},
    {Voice: '300m', Data: "5"},
    {Voice: '350m', Data: "12.7"},
    {Voice: '375m', Data: "8"},
    {Voice: '400m', Data: "17"},
    {Voice: '425m', Data: "15"},
    {Voice: '450m', Data: "12"},
    {Voice: '500m', Data: "6"}
];
var settings = {
    title: "",
    borderLineWidth: 1,
    enableAnimations: true,
    backgroundColor:"transparent",
    showBorderLine: false,
    description: '',
    showLegend: false,
    showToolTips: false,
    source: sampleData,
    xAxis: {
        dataField: 'Voice',
        unitInterval: 3,
        verticalTextAlignment:'bottom',
        formatFunction: function (value) {
        return value;
        },
        labels: {
            angle: 0,
            offset: {x: 0, y: 24}
        },
        gridLines: {
            visible: false,
            color: '#FFFFFF'
        },
        title:{
            text: 'Voice usage',
            offset: { x: 0, y: 27 }
        },
        tickMarks: {
            visible: false,
        },
        // make the vertical line dispaly none
        valuesOnTicks: false
    },
    colorScheme: 'scheme05',
    seriesGroups: [{
        type: 'column',
        valueAxis: {
        position: 'right',
        title: {
            text: 'Data usage',
            offset: { x: 25, y: 0 }
        },
        formatFunction: function (value) {
            return value + '%';
        },
        labels: {
            horizontalAlignment: 'left',
            offset: {x: 25, y: 0}
        },
        line: {visible: false},
        maxValue: 18,
        minValue: 0,
        unitInterval: 2
        },
        series: [{
            legendFillColor:"transparent",
            legendLineColor:"transparent",
            dataField: 'Data',
            displayText: 'Voice usage',
            toolTipFormatFunction: function (value, itemIndex, serie, group, categoryValue) {
                return '<div class="tooltip-inner" style="text-align:left;z-index: 2";><b>Year-Voice: </b>' + categoryValue
                + '<br /><b>GDP Data:</b> ' + value + ' %</div>'
                }
            }
        ]}
    ]
};
// prepare chart data
// select the chartContainer DIV element and render the chart.
$('#chartContainer').jqxChart(settings);
