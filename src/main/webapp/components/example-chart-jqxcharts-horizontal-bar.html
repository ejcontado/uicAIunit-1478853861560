Horizontal Bar Chart (jqWidget)


JS & CSS dependency:

For bar chart(jqWidget) components, it will be ok to import following css file:
<link  href="../wulfdist/css/jqx.base.css" rel="stylesheet"/>
<link href="../wulfdist/css/wulf.basic.min.css" rel="stylesheet"/>
<link href="../wulfdist/css/wulf.chart.min.css" rel="stylesheet"/>
<link href="../wulfdist/css/wulf.panel.min.css" rel="stylesheet"/>

Place the chart into a <div> 


<!-- Main chart div -->
<div id="chartContainer" style="width:800px; height:300px">
</div>

Data and Settings ( For more settings, see the jqxChart API Reference )


// the data to show on the chart
var sampleData = [
    { Phone: 'iPhone 3', Percentage: 6.4 },
    { Phone: 'Nokia N9', Percentage: 4.2 },
    { Phone: 'Blackberry', Percentage: 4.6 },
    { Phone: 'iPhone 4', Percentage: 3.5 },
    { Phone: 'HTC Desire', Percentage: 2.9 }];
// prepare jqxChart settings
var settings = {
    title: "",
    description: "",
    borderLineWidth: 1,
    backgroundColor:"transparent",
    showBorderLine: false,
    showLegend: true,
    legendLayout: {
        left: 600,
        top: 123,
        width: 300,
        height: 200,
        flow: 'vertical'
    },
    enableAnimations: true,
    padding: { left: 82, top: 200, right: 240, bottom: 36 },
    source: sampleData,
    xAxis:
    {
        dataField: 'Phone',
        flip: false,
        labels: { 
            visible: true, 
            angle: 90,
            offset: {x: -82, y: 0},
            horizontalAlignment: "left",
        },
        gridLines: {
            visible: false,
            color: '#FFFFFF'
        },
        tickMarks: {
            visible: false,
        },
    },
    valueAxis:
    {
        flip: true,
        labels: {
            visible: true,
            formatFunction: function (value) {
                return value
            },
            offset:{x:0, y:4}
        },
        gridLines: {
            visible: true,
            step:1,
            color: '#B1B1B1',
            lineWidth: 10,
        },
        line:{visible: false},
        minValue: 0,
        maxValue: 7,
        unitInterval: 1,
        padding:{left:58,  right:0 },
        tickMarks :{
            size: 20,
        },
    },
    colorScheme: 'scheme01',
    seriesGroups:[
       {
           type: 'column',
           orientation: 'horizontal',
           columnsGapPercent: 13,
           toolTipFormatSettings: { thousandsSeparator: ',' },
           toolTipFormatFunction: function(value){
               return "Data "+ value +"%"
           },
           series: [
               { dataField: 'Percentage',
                   displayText: 'Voice' ,
                   formatSettings: { sufix: '%'},
                   labels: {
                        visible: true,
                        formatSettings: { sufix: '%'},
                        offset: { x: -30, y: 2 }
                    },
               },
           ]
       }
    ]
};
// setup the chart
$('#chartContainer').jqxChart(settings);
