Data-placement can be: top | bottom | left | right + also empty + optionally also auto to allow automatic positioning on the opposite side near screen edges. Title should not be used in case of popovers (OT3 does not provide it).
You can control the arrow location for example with .popover.right .arrow { top: 22% }

    <script>
        $('[class="popover-icon"]').popover({
        html: true,
        content: function () {
            return $(jQuery(this).data('target-selector')).html();
        }
        }).on('shown.bs.popover', function (e) {
            var popover = jQuery(this);
            jQuery(this).parent().find('div.popover .n-close').on('click', function (e) {
                popover.popover('hide');
            });
        });
    </script>

    <!-- balloon fadeout feature-->

    <!-- JS-script needed  -->
    <script>
        $('.fadeout').on('shown.bs.popover', function () {
            var id="#"+$(this).attr("aria-describedby");

            //the first 1000 means how long the balloon fadeout take, the second 1000 means how long when balloon start fadeout after shown, any number value can be apply
            setTimeout(function(){
                $(id).fadeOut(1000, $.proxy(function(){
                    $(this).popover('hide');
                    }))
            },1000);
    });
    </script>
