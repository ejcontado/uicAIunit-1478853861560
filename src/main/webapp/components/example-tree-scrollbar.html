        To enable scorllbar for tree, you can simply call enableScrollbar method for tree component. The parameters stand for the width and height of the tree component. 
        
<!-- Initiate tree data in js -->
<script>
$(document).ready(
 function ()
 {
	$('#myTree').tree({
		dataSource: dataSource,
		folderSelect :false
	});

	<-- get data for tree component -->
	dataSource = function(parentData, callback){//...}
	$('#myTree').tree('enableScrollbar', 250, 300);
 }
</script>

        To disable the tree, you can simply call disable method for tree component 
        
<script>
$(document).ready( function ()
{
	$('#myTree').tree('disable');
});
</script>
